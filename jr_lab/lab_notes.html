import matplotlib.pyplot as plt
import numpy as np
def calc_em(i, v, r) -> float:
    return 2 * v / ((r)**2 * (7.8e-4 * i)**2)

def calc_em_error(i, i_unc, v, v_unc, r, r_unc):
    c = 7.8e-4
    var = (2/(r * i * c)**2)**2 * v_unc**2 + (4*v/(r**2 * i**3 * c**2))**2 * i_unc**2 + (4 * v /(r**3 * i**2 * c**2)**2) * r_unc**2 
    return np.sqrt(var)
